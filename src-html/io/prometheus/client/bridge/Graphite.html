001package io.prometheus.client.bridge;
002
003import io.prometheus.client.Collector;
004import io.prometheus.client.CollectorRegistry;
005
006import java.io.BufferedWriter;
007import java.io.IOException;
008import java.io.PrintWriter;
009import java.net.Socket;
010import java.util.Collections;
011import java.util.logging.Level;
012import java.util.logging.Logger;
013import java.util.regex.Pattern;
014import java.util.regex.Matcher;
015
016/**
017 * Export metrics in the Graphite plaintext format.
018 * <p>
019 * <pre>
020 * {@code
021 *  Graphite g = new Graphite("localhost", 2003);
022 *  // Push the default registry once.
023 *  g.push(CollectorRegistry.defaultRegistry);
024 *
025 *  // Push the default registry every 60 seconds.
026 *  Thread thread = g.start(CollectorRegistry.defaultRegistry, 60);
027 *  // Stop pushing.
028 *  thread.interrupt();
029 *  thread.join();
030 * }
031 * </pre>
032 * <p>
033 */
034public class Graphite {
035  private static final Logger logger = Logger.getLogger(Graphite.class.getName());
036
037  private final String host;
038  private final int port;
039  private static final Pattern INVALID_GRAPHITE_CHARS = Pattern.compile("[^a-zA-Z0-9_-]");
040  /**
041   * Construct a Graphite Bridge with the given host:port.
042   */
043  public Graphite(String host, int port) {
044    this.host = host;
045    this.port = port;
046  }
047
048  /**
049   * Push samples from the given registry to Graphite.
050   */
051  public void push(CollectorRegistry registry) throws IOException {
052    Socket s = new Socket(host, port);
053    BufferedWriter writer = new BufferedWriter(new PrintWriter(s.getOutputStream()));
054    Matcher m = INVALID_GRAPHITE_CHARS.matcher("");
055    long now = System.currentTimeMillis() / 1000;
056    for (Collector.MetricFamilySamples metricFamilySamples: Collections.list(registry.metricFamilySamples())) {
057      for (Collector.MetricFamilySamples.Sample sample: metricFamilySamples.samples) {
058        m.reset(sample.name);
059        writer.write(m.replaceAll("_"));
060        for (int i = 0; i < sample.labelNames.size(); ++i) {
061          m.reset(sample.labelValues.get(i));
062          writer.write("." + sample.labelNames.get(i) + "." + m.replaceAll("_"));
063        }
064        writer.write(" " + sample.value + " " + now + "\n");
065      }
066    }
067    writer.close();
068    s.close();
069  }
070
071  /**
072   * Push samples from the given registry to Graphite every minute.
073   */
074  public Thread start(CollectorRegistry registry) {
075    return start(registry, 60);
076  }
077
078  /**
079   * Push samples from the given registry to Graphite at the given interval.
080   */
081  public Thread start(CollectorRegistry registry, int intervalSeconds) {
082    Thread thread = new PushThread(registry, intervalSeconds);
083    thread.setDaemon(true);
084    thread.start();
085    return thread;
086  }
087
088  private class PushThread extends Thread {
089    private final CollectorRegistry registry;
090    private final int intervalSeconds;
091
092    PushThread(CollectorRegistry registry, int intervalSeconds) {
093      this.registry = registry;
094      this.intervalSeconds = intervalSeconds;
095    }
096
097    public void run() {
098      long waitUntil = System.currentTimeMillis();
099      while (true) {
100        try {
101          push(registry);
102        } catch (IOException e) {
103          logger.log(Level.WARNING, "Exception " + e + " pushing to " + host + ":" + port, e);
104        }
105
106        long now = System.currentTimeMillis();
107        // We may skip some pushes if we're falling behind.
108        while (now >= waitUntil) {
109          waitUntil += intervalSeconds * 1000;
110        }
111        try {
112          Thread.sleep(waitUntil - now);
113        } catch (InterruptedException e) {
114          return;
115        }
116      }
117    }
118  }
119}
